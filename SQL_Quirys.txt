##GROUP BY and Numeric Function(aggrigate Function)##
##In other Words Group By Clause##
......................................................

SELECT SalesOrderId, UnitPrice
FROM sales.salesorderdetail;
 
SELECT SalesOrderId, SUM(UnitPrice) AS TotalUnitPricePerSales
FROM sales.salesorderdetail
GROUP BY SalesOrderId
 
SELECT SalesOrderId, AVG(UnitPrice) AS AvgUnitPricePerSales
FROM sales.salesorderdetail
GROUP BY SalesOrderId
 
SELECT SalesOrderId, COUNT(UnitPrice) AS CountUnitPrice
FROM sales.salesorderdetail
GROUP BY SalesOrderId
 
SELECT SalesOrderId, MAX(UnitPrice) AS MaxUnitPrice
FROM sales.salesorderdetail
GROUP BY SalesOrderId
 
SELECT SalesOrderId, MIN(UnitPrice) AS MaxUnitPrice
FROM sales.salesorderdetail
GROUP BY SalesOrderId

##Having Clause##
..................

SELECT SalesOrderId, SUM(UnitPrice) AS TotalUnitPricePerSales
FROM sales.salesorderdetail
WHERE SalesOrderID > 50000
GROUP BY SalesOrderId
HAVING SUM(UnitPrice) > 10000
ORDER BY SalesOrderId DESC

Note: We have to maintain same order while writting sql It is mendetory,
For sql Server the order is: FROM -> WHERE -> GROUP BY -> SELECT -> ORDER BY.
